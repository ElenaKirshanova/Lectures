			\item Если $h(x)  =f(x+v)$, то 
			\[
				\hat{h}(y) = \hat{f}(y) \cdot e^{2\pi i \ScProd{v}{y}} \quad \forall v \in \RR^n.
			\]
			\item Если $h(x) = f(x) e^{2\pi i \ScProd{v}{x}}$, то 
			\[
				\hat{h}(y) = \hat{f}(y - v).
			\]
		\end{enumerate}
